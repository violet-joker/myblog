# lazyvim

```shell
:<leader>uh # 开启/关闭内嵌提示信息
ctrl+n/p    # 切换上下候选词
gc          # 注释选中的代码
za          # 切换当前代码块的折叠状态
zr          # 展开所有代码块
zm          # 折叠所有代码块
m+字符      # 设置标记
'+字符      # 跳转到标记
:delmarks!  # 删除所有标记
bd          # 关闭当前页面
Shift + h/l # 前后页面切换
```


# niri

```shell
Mod + [/]           # 当前窗口吸入到右侧/左侧
Mod + Shift + -/+   # 纵向调整窗口大小
Mod + -/+           # 横向调整窗口大小
Mod + h/j/k/l       # 移动聚焦点
Mod + Ctrl + h/l    # 移动当前列位置
Mod + Ctrl + j/k    # 移动当前行位置(同列中移动)
```


# linux

```shell
sudo -                      # 切换为root用户
sudo adduser xf             # 创建用户
sudo usermod -aG sudo xf    # 提升管理员权限(允许使用sudo命令)
sudo userdel -r user        # 删除用户(先确保不是以user身份登陆)
sudo pkill -u user          # 切换账户可能存在残留的守护进程
id user                     # 查看制定用户的id
cut -d: -f1 /etc/passwd     # 查看用户列表
hostnamectl                 # 查看内核及系统概况
ls -lh                      # 查看文件属性
loginctl                    # 查看会话属性
loginctl show-session id -p Type
```

# git

```shell
git config --global user.name "name"
git config --global user.email "email"
ssh-keygen -t ed25519 -C "email@example.com"
git branch                      # 查看分支
git branch -m newName           # 修改分支名
git checkout branchName         # 切换到已存在的分支
git chechout -b newBranch       # 创建并切换到新的分支
git checkout -f branchName      # 强制切换分支(不保存未提交内容)
git status                      # 查看改动的文件
git add .
git reset                       # 清除add的内容
git commit -m "comment"
git push
git psuh origin HEAD            # 推送至当前分支
git psuh origin --delete name   # 删除分支
git checkout .                  # 一键重置到上一次commit版本
git clean -fd                   # 强制删除未跟踪的文件和目录
```
