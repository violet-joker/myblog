# lazyvim

```bash
:<leader>uh # 开启/关闭内嵌提示信息
ctrl+n/p    # 切换上下候选词
gc          # 注释选中的代码
za          # 切换当前代码块的折叠状态
zr          # 展开所有代码块
zm          # 折叠所有代码块
m+字符      # 设置标记
'+字符      # 跳转到标记
:delmarks!  # 删除所有标记
bd          # 关闭当前页面
Shift + h/l # 前后页面切换
```

# linux

```bash

/usr                        # 由包管理器管理
ldconfig                    # 手动执行以刷新库依赖(安装新库时)
```

# f3(测试硬盘是否扩容)

```bash
# 查看挂在的磁盘
df -h

# 先卸载(例如df -h得知u盘识别为sda)
sudo umount /dev/sda1

# destructive会覆盖数据，用于检测空卡容量
sudo f3probe --destructive /dev/sda
sudo f3probe --destructive --time-ops /dev/sda

# 若测得结果为扩容卡，使用f3fix修复

# 格式化失败，通常是因为扩容卡分区表已损坏，
# 或内核由于之前的读写错误锁定了该设备，
# 擦除所有已知分区签名，修复与旧分区表冲突导致的格式化失败:
sudo wipefs -a /dev/sda
sudo mkfs.vfat -I /dev/sda
```

# ffmpeg

```bash
# 将mp3转换成wav,并设置单声道
ffmpeg -i input.mp3 -ac 1 output.wav
# 可选参数
-sample_fmt u8
-c:a pcm_u8

# 转换视频
# 将mp4文件转换成便于esp32解析的avi文件
ffmpeg -i input.mp4 -s 320x240 -c:v mjpeg -q:v 10 -ac 1 -f avi output.avi
-an  # 剔除音频
-r   # 设置帧率
-vsync cfr  # 强制恒定帧率，避免帧丢失、帧重复

# 合并video.m4s和audio.m4s
ffmpeg -i video.mp4 -i audio.mp4 -codec copy output.mp4

# 视频切片
ffmpeg -i input.mp4 -ss 00:00:00 -to 03:48:55 -c copy output.mp4
```
