# 帧动画

基础帧动画节点如下，特殊动作可继承节点，重写enter、update、exit等方法。
在主循环/状态机中调用update和render函数更新帧动画。

```c++
class Node {
public:
    Node(SDL_Renderer *renderer, const char *path, float w, float h, int ms, int totalFrames)
    : frameWidth(w), frameHeight(h), frameDelay(ms), totalFrames(totalFrames)
    {
        SDL_Surface *surface = IMG_Load(path);
        spriteSheet = SDL_CreateTextureFromSurface(renderer, surface);
        currentFrame = 0;
        lastUpdateTime = SDL_GetTicks();
    }

    virtual void enter() {}

    virtual void update() {
        Uin32 currentTime = SDL_GetTicks();
        if (currentTime - lastUpdateTime >= frameDelay) {
            currentFrame = (currentFrame + 1) % totalFrames;
            lastUpdateTime = currentTime;
        }
    }

    virtual void exit() {}

    virtual void render(SDL_Renderer *renderer, float x, float y) {
        SDL_FRect srcRect = {
            currentFrame * frameWidth,
            0,
            frameWidth, 
            frameHeight
        };
        SDL_FRect destRect = {
            x, y,
            frameWidth, frameHeight
        };
        SDL_RenderTexture(renderer, spriteSheet, &srcRect, &destRect);
    }

private:
    SDL_Texture *spriteSheet;
    float frameWidth;
    float frameHeight;
    int totalFrames;
    int currentFrame;
    Uint32 frameDelay;
    Uint32 lastUpdateTime;
};
```

# 状态机

状态机管理各个动作之间的切换

```c++
class StateMachine {
public:
    void update() {
        if (current_state)
            current_state->update();
    }

    void render(SDL_Renderer *renderer, float x, float y) {
        current_state->render(renderer, x, y);
    }

    // 用于注册完所有动作后设置初始动作
    void set_entry(const std::string& state) {
        current_state = state_pool[state];
    }

    // 动作切换，终止当前动作，进入新动作
    void switch_to(const std::string& state) {
        if (current_state) current_state->exit();
        current_state = state_pool[state];
        if (current_state) current_state->enter();
    }

    // 初始化阶段注册所有动作
    void register_state(const std::string& state, StateNode *stateNode) {
        state_pool[state] = stateNode;
    }
private:
    StateNode *current_state;
    std::map<std::string, StateNode*> state_pool;
};
```
