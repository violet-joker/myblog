# 8266配网

> AP模式: wifi模块产生热点，提供无线接入服务(TCP Server)。

> STA模式: wifi模块作为终端(站点)，可以连接到AP(TCP Client)。

```c++
#include <ESP8266WiFi.h>
#include <DNSServer.h>
#include <ESP8266WebServer.h>
#include <WiFiManager.h>

void config_wifi() {
    WiFiManager wifiManager;
    // wifiManager.resetSettings();
    // Serial.println("wifi settings cleared");

    // 尝试连接5秒，超时后进入配网
    wifiManager.setConnectTimeout(5);
    // 设置AP接入点, 访问192.168.4.1
    wifiManager.autoConnect("8266AP");
    // 设置带密码的WIFI接入点
    // wifiManager.autoConnect("8266AP", "123456");
    // 手机或电脑连接AP接入点，再配置想要8266连接的wifi
    Serial.print("8266 connected to ");
    Serial.println(WiFi.SSID());
    Serial.println("IP address:");
    Serial.println(WiFi.localIP());
}
```

# 8266网络通信

## UDP通信

```c++
#include <ESP8266WiFi.h>
#include <WiFiUdp.h>

#define localUdpPort 1234
#define remoteUdpPort 4321
#define BUF_SIZE 255

WiFiUDP Udp;
char buf[BUF_SIZE];

void setup() {
    // ... 配网 ...
    // 启动udp监听服务
    if (Udp.begin(localUdpPort)) {
        Serial.printf("当前IP:%s, UDP端口:%d\n",
        WiFi.localIP().toString().c_str(),
        localUdpPort);
    } else {
        Serial.println("监听失败");
    }
}

void loop() {
    // 获得解析包
    int packetSize = Udp.parsePacket();
    if (packetSize) {
        Serial.printf("从%s:%d收到数据包字节数:%d\n",
                Udp.remoteIP().toString().c_str(),
                Udp.remotePort(),
                packetSize);

        int len = Udp.read(buf, BUF_SIZE);
        if (len > 0) {
            buf[len] = 0;
            Serial.printf("UDP数据为:%s\n", buf);
        }

        // 传递数据
        // 配置远端ip地址和端口
        Udp.beginPacket(Udp.remoteIP(), remoteUdpPort);
        // 将数据写入发送缓冲区
        Udp.write(buf);
        // 发送数据
        Udp.endPacket();
    }
    delay(10);
}

```

# 连接tft屏

SDA连接D7，SCL连接D5；CS，RST，DC任意

```c++
#include <Adafruit_GFX.h>
#include <Adafruit_ST7735.h>
#include <SPI.h>
#include <U8g2_for_Adafruit_GFX.h>

#define D0 16
#define D1 5
#define D2 4

#define TFT_CS  D0
#define TFT_DC  D1
#define TFT_RST D2

Adafruit_ST7735 tft = Adafruit_ST7735(TFT_CS, TFT_DC, TFT_RST);

void setup() {
  Serial.begin(9600);
  tft.initR(INITR_BLACKTAB);
  tft.setRotation(1);
  tft.fillScreen(ST7735_BLACK);

  tft.setCursor(10, 10);
  tft.setTextColor(ST7735_BLUE);
  tft.setTextSize(1);
  tft.println("hello world!");
}

void loop() {
  delay(1000);
}

```
